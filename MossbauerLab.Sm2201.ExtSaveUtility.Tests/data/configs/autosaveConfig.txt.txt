# ВНИМАНИЕ!!!!!!! с символа решетка (№) начинаются комментарии
# Настройки идут в виде пар ключ = значение
# В конфигурации должны присутствовать ВСЕ пары значений, а именно следующие ключи: 
#    - state, 
#    - useChannel1, 
#    - useChannel2,
#    - channel1Period
#    - channel2Period
#    - outputDir
#    - archiveDir
# В ЭТОЙ версии все значения чувтсвительны к регистру, ПОЭТОМУ НЕОБХОДИМО использовать их так как они 
# перечислены выше
# state - состояние автосохранения: 0 - ВЫКЛЮЧЕНО, 1 - ВКЛЮЧЕНО, например, state = 0
state = 1
# useChannel1 - сохранение спектров по каналу 1, 0 - ВЫКЛЮЧЕНО, 1 - ВКЛЮЧЕНО, например, useChannel1 = 1
useChannel1 = 1
# useChannel2 - сохранение спектров по каналу 2, 0 - ВЫКЛЮЧЕНО, 1 - ВКЛЮЧЕНО, например, useChannel2 = 1
useChannel2 = 1
# channel1Period - период автосохранения спектров по каналу 1 в СЕКУНДАХ, например, для сохранения раз в час, channel1Period = 3600
channel1Period = 3600
# channel2Period - период автосохранения спектров по каналу 2 в СЕКУНДАХ,например, для сохранения раз в час, channel2Period = 3600
channel2Period = 3600
# outputDir - директория (ПОЛНЫЙ ПУТЬ), в которую, SM2201 сохраняет спектры, например outputDir=C:\\SM2201\\Specs
outputDir=C:\\SM2201\\Specs
# archiveDir - директория (ПОЛНЫЙ ПУТЬ), в которую спектры будут копироваться из outputDir с уникальным именем, содержащим timestamp
archiveDir=C:\\SavedSpecs
